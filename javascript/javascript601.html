03. 배열 메서드 : unshift() / push() : 배열 요소에 추가하기 :반환 (숫자)

번호 기본 값 메서드 메서드 값 결과값 
{
    const arr = [100, 200, 300 ,400, 500];
    const text = arr.push(600);

    document.querySelector(".sample03_N1").innerHTML = "1";
    document.querySelector(".sample03_Q1").innerHTML = "[100, 200, 300 ,400, 500]";
    document.querySelector(".sample03_M1").innerHTML = ".push(600)";
    document.querySelector(".sample03_M1").innerHTML = text;
    document.querySelector(".sample03_R1").innerHTML = arr;

    const arr2 = [100, 200, 300 ,400, 500];
    const text2 = arr2.unshift(600);

    document.querySelector(".sample03_N2").innerHTML = "2";
    document.querySelector(".sample03_Q2").innerHTML = "[100, 200, 300 ,400, 500]";
    document.querySelector(".sample03_M2").innerHTML = ".unshift(600)";
    document.querySelector(".sample03_P2").innerHTML = text2;
    document.querySelector(".sample03_R2").innerHTML = arr2;
}

번호 기본 값 메서드 결과값 
{
    const arr = [100, 200, 300 ,400, 500];
    const text = arr.shift();

    document.querySelector(".sample04_Q1").innerHTML = "[100, 200, 300 ,400, 500]";
    document.querySelector(".sample04_M1").innerHTML = ".shift()";
    document.querySelector(".sample04_P1").innerHTML = text;
    document.querySelector(".sample04_R1").innerHTML = arr;

    document.querySelector(".sample04_M2").innerHTML = ".pop()";
    document.querySelector(".sample04_P2").innerHTML = text2;
    document.querySelector(".sample04_R2").innerHTML = arr2;
}
