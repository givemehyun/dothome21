04. 1부터 100까지 출력하기 : 짝수는 빨간색, 홀수는 파란색 출력 

{
    document.write ("***-4. 1부터 100까지 출력 (짝수는 빨간색, 홀수는 파란색 출력 ***
")
    for(let i=1; i<=100; i++){
        if ( i % 2 == 0){
            document.write("<span style='color:red'>" + i + "</span>");
        } else {
            document.write("<span style='color:blue'>" + i + "</span>");
        }
    }
}

06. 구구단 출력하기 

{
    document.write("*** 06.구구단 출력하기 ***");
    //2~9 for (var i=2; i<=9; i++)
    //1~9 for (var i=1; i<=9; i++)

    // num1 * num2 = sum
    // 2 * 1 = 2;          3 * 1 = 3;
    // 2 * 2 = 4;          3 * 2 = 6;
    // 2 * 3 = 6;          3 * 3 = 9;
    // 2 * 4 = 8;          ..........
    // 2 * 5 = 10;         9 * 1 = 9
    // 2 * 6 = 12;         9 * 2 = 18
    // 2 * 7 = 14;
    // 2 * 8 = 16;
    // 2 * 9 = 18;

    for (var i = 2; i <= 9; i++) {
        for (var j = 1; j <= 9; j++) {
            var sum = i * j;
            document.write(i + " * " + j + " = " + sum );
        }
    }
}

07. 테이블 만들기1 

{
    document.write("*** 07. 테이블 만들기1 *** ");

    let num = 1;
    let table = "<table>";

    for( let i=1; i<=10; i++ ){
        table += "<tr>";
        for( let j=1; j<=10; j++ ){
            table += "<td>"+ num +"</td>";
            num++;
        }
        table += "</tr>";
    }

    table += "</table>";

    document.write(table);
}

08. 테이블 만들기2
