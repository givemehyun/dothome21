}

11. 문자열 메서드 : encodeURL() : 문자열을 인코딩 : 반환(문자열)

번호 기본값 메서드 결과값 
{
    const text = "https://givemehyun.co.kr/자바스크립트.html";
    const url = encodeURI(text);

    document.querySelector(".sample11_R1").innerHTML = url;

    const text2 = "https://givemehyun.co.kr/자바스크립트.html";
    const url2 = encodeURIComponent(text2);

    document.querySelector(".sample11_R2").innerHTML = url2;

    const text3 = "https://givemehyun.co.kr/%EC%9E%90%EB%B0%94%EC%8A%A4%ED%81%AC%EB%A6%BD%ED%8A%B8.html";
    const url3 = decodeURI(text3);

    document.querySelector(".sample11_R3").innerHTML = url3;

    const text4 = "https%3A%2F%2Fgivemehyun.co.kr%2F%EC%9E%90%EB%B0%94%EC%8A%A4%ED%81%AC%EB%A6%BD%ED%8A%B8.html";
    const url4 = decodeURIComponent(text4);

    document.querySelector(".sample11_R4").innerHTML = url4;
}
