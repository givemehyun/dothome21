02. 날짜 메서드

번호 기본값 메서드 결과값 
{
    const date = new Date(); 
    const locale = date.toLocaleString();
    
    document.querySelector(".sample02_R1").innerHTML = locale;

    const localeDate = date.toLocaleDateString();

    document.querySelector(".sample02_R2").innerHTML = localeDate;

    const localeTime =  date.toLocaleTimeString();

    document.querySelector(".sample02_R3").innerHTML = localeTime;
}

03. 날짜 메서드 : Date.parse() : 날짜 문자열 타임스탬프 불러오기

번호 기본값 메서드 결과값 
{
    const date = Date.parse("2021/8/11");

    document.querySelector(".sample03_N1").innerHTML = "1";
    document.querySelector(".sample03_Q1").innerHTML = "2021/8/11";
    document.querySelector(".sample03_M1").innerHTML = "Date.parse()";
    document.querySelector(".sample03_R1").innerHTML = date;

    const date2 = Date.parse("8 11 2021");

    document.querySelector(".sample03_N2").innerHTML = "2";
    document.querySelector(".sample03_Q2").innerHTML = "11 8 2021";
    document.querySelector(".sample03_M2").innerHTML = "Date.parse()";
    document.querySelector(".sample03_R2").innerHTML = date2;

    const date3 = Date.now();

    document.querySelector(".sample03_N3").innerHTML = "3";
    document.querySelector(".sample03_Q3").innerHTML = "";
    document.querySelector(".sample03_M3").innerHTML = "Date.now()";
    document.querySelector(".sample03_R3").innerHTML = date3;
}
