18. 콜백함수 : 반복문 

{
    document.write("***18. 콜백함수 : 반복문   ***");

    function func(){
        document.write("함수가 실행되었습니다.");
    }
    function callback(str){
        for(let i=1; i<=5; i++){
            str(i);
        }
    }
    callback(func);
}
