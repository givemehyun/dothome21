02. 수학 메서드 : Math.random()

번호 기본값 메서드 결과값 
{
    document.querySelector(".sample02_N1").innerHTML ="1";
    document.querySelector(".sample02_Q1").innerHTML ="0~1";
    document.querySelector(".sample02_M1").innerHTML =" Math.random()";
    document.querySelector(".sample02_R1").innerHTML = Math.random();

    document.querySelector(".sample02_B1 button").addEventListener("click", function(){
        document.querySelector(".sample02_R1").innerHTML = Math.random();
    });
    //버튼을 클릭하면 경고창을 띄워주세요 !

    document.querySelector(".sample02_N2").innerHTML ="2";
    document.querySelector(".sample02_Q2").innerHTML ="0~9";
    document.querySelector(".sample02_M2").innerHTML =" Math.random()";
    document.querySelector(".sample02_R2").innerHTML = Math.floor(Math.random()*10);

    document.querySelector(".sample02_B2 button").addEventListener("click", function(){
        document.querySelector(".sample02_R2").innerHTML = Math.floor(Math.random()*10);
    });

    document.querySelector(".sample02_N3").innerHTML ="3";
    document.querySelector(".sample02_Q3").innerHTML ="10~19";
    document.querySelector(".sample02_M3").innerHTML =" Math.random()";
    document.querySelector(".sample02_R3").innerHTML = Math.floor(Math.random()*10+10);

    document.querySelector(".sample02_B3 button").addEventListener("click", function(){
        document.querySelector(".sample02_R3").innerHTML = Math.floor(Math.random()*10+10);
    });
}
